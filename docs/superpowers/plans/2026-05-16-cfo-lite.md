# Phase 4.0 — CFO Lite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add order-level profitability snapshots, a CFO Lite tab inside `/profitability`, an extended CFO agent scan with order-level exceptions, and a profitability chat intent — all read-only against existing billing and inventory data.

**Architecture:** Two new DB objects (`cost_price` column on `catalog_items` + `profitability_snapshots` table) back server-side analytical snapshots. A generation API populates them; the CFO tab reads them via the Supabase client. The existing CFO agent scan is extended to detect order-level exceptions and generate snapshots during its run. The chat engine gains a `profitability` intent that queries snapshots. No billing, invoice, or inventory-logic code is touched.

**Tech Stack:** Next.js 16 App Router, TypeScript, Supabase (service role + anon RLS), Tailwind CSS v4, RTL Hebrew UI, `@supabase/supabase-js`

**Spec:** `docs/superpowers/specs/2026-05-16-cfo-lite-design.md`

```sql
  customer_id           text        DEFAULT NULL,

  material_cost         numeric     NOT NULL DEFAULT 0,
  vehicle_cost          numeric     NOT NULL DEFAULT 0,

  other_cost            numeric     NOT NULL DEFAULT 0,
  overhead_cost         numeric     NOT NULL DEFAULT 0,
  total_cost            numeric     NOT NULL DEFAULT 0,
  gross_profit          numeric     NOT NULL DEFAULT 0,
  gross_margin_percent  numeric     NOT NULL DEFAULT 0,
  confidence_level      text        NOT NULL DEFAULT 'missing_data'
                        CHECK (confidence_level IN ('high', 'medium', 'low', 'missing_data')),
  missing_data          jsonb       NOT NULL DEFAULT '[]',
  source_data           jsonb       NOT NULL DEFAULT '{}',
  created_at            timestamptz NOT NULL DEFAULT now(),
  updated_at            timestamptz NOT NULL DEFAULT now()
);

-- Unique: one order-level snapshot per order (work_diary_id IS NULL = order-level)
CREATE UNIQUE INDEX IF NOT EXISTS uq_profitability_snapshot_order
  ON public.profitability_snapshots (order_id)
  WHERE work_diary_id IS NULL;

-- RLS: anon can read; service_role has full access
ALTER TABLE public.profitability_snapshots ENABLE ROW LEVEL SECURITY;

DO $$ BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_policies
    WHERE tablename = 'profitability_snapshots' AND policyname = 'anon read snapshots'
  ) THEN
    EXECUTE 'CREATE POLICY "anon read snapshots" ON public.profitability_snapshots
      FOR SELECT TO anon USING (true)';
  END IF;
END $$;
```

```typescript
// ─── Order-level profitability ───────────────────────────────────────────────
// Read-only integration with work_diaries + inventory_consumptions.
// Does not modify billing, accounting, or inventory data.

export type ConfidenceLevel = "high" | "medium" | "low" | "missing_data";

export type MissingDataTag =
  | "no_revenue"
  | "no_linked_diaries"
  | "no_crew_data"
  | "missing_cost_price"
  | "no_material_cost"
  | "no_vehicle_data"
  | "no_approved_diary";

export interface OrderInput {
  id: string;
  orderNumber: string;
  customer: string;
  billedAmount?: number | null;
}

export interface OrderDiaryInput {
  id: string;
  orderId?: string | null;
  billedAmount?: number | null;
  materialCost?: number | null;
  equipmentCost?: number | null;
  vehicleCostOverride?: number | null;
  vehicleNumber?: string | null;
  crewLeaderName?: string | null;
  crewMembers: string[];
  approvalStatus?: string | null;
  isBillable?: boolean | null;
}

export interface OrderConsumptionInput {
  id: string;
  orderId: string;
  quantity: number;
  costPrice: number | null; // null = cost_price unknown for this item
}

export interface OrderProfitabilityResult {
  orderId: string;
  orderNumber: string;
  customerName: string;
  revenue: number;
  laborCost: number;
  materialCost: number;
  vehicleCost: number;
  equipmentCost: number;
  subcontractorCost: number;
  otherCost: number;
  overheadCost: number;
  totalCost: number;
  grossProfit: number;
  grossMarginPercent: number;
  confidenceLevel: ConfidenceLevel;
  missingData: MissingDataTag[];
  sourceData: {
    diaryIds: string[];
    consumptionIds: string[];
    orderId: string;
  };
}

export const CONFIDENCE_LABELS: Record<ConfidenceLevel, string> = {
  high:         "גבוהה",
  medium:       "בינונית",
  low:          "נמוכה",
  missing_data: "חסרים נתונים",
};

export const CONFIDENCE_COLORS: Record<ConfidenceLevel, string> = {
  high:         "bg-green-100 text-green-800",
  medium:       "bg-amber-100 text-amber-800",
  low:          "bg-orange-100 text-orange-800",
  missing_data: "bg-gray-100 text-gray-500",
};

export const MISSING_DATA_LABELS: Record<MissingDataTag, string> = {
  no_revenue:         "חסרה הכנסה",
  no_linked_diaries:  "אין יומנים מקושרים",
  no_crew_data:       "אין נתוני צוות",
  missing_cost_price: "חסר מחיר עלות",
  no_material_cost:   "חסרה עלות חומרים",
  no_vehicle_data:    "אין נתוני רכב",
  no_approved_diary:  "אין יומן מאושר",
};

export function calculateOrderProfitability(
  order: OrderInput,
  allDiaries: OrderDiaryInput[],
  rates: CostRates,
  allConsumptions: OrderConsumptionInput[]
): OrderProfitabilityResult {
  const diaries = allDiaries.filter(d => d.orderId === order.id && d.isBillable !== false);
  const consumptions = allConsumptions.filter(c => c.orderId === order.id);

  // Revenue: order.billedAmount takes precedence; fall back to sum of diary amounts
  const revenue =
    order.billedAmount != null && order.billedAmount > 0
      ? order.billedAmount
      : diaries.reduce((s, d) => s + (d.billedAmount ?? 0), 0);

  // Per-diary cost rollup
  let laborCost = 0;
  let vehicleCost = 0;
  let equipmentCost = 0;
  let materialCostDiaries = 0;
  let directForOverhead = 0;
  let totalWorkers = 0;
  let someHaveVehicle = false;
  let hasApproved = false;

  for (const d of diaries) {
    const crewCount = d.crewMembers.filter(m => m.trim()).length;
    const hasLeader = d.crewLeaderName?.trim() ? 1 : 0;
    totalWorkers += crewCount + hasLeader;

    const dLabor = hasLeader * rates.teamLeaderDailyCost + crewCount * rates.workerDailyCost;
    const hasVeh = d.vehicleNumber?.trim() ? 1 : 0;
    if (hasVeh) someHaveVehicle = true;
    const dVehicle = d.vehicleCostOverride != null
      ? d.vehicleCostOverride
      : hasVeh ? (rates.vehicleDailyCost + rates.fuelCostPerDay) : 0;
    const dEquip = d.equipmentCost ?? 0;
    const dMaterial = d.materialCost ?? 0;

    laborCost += dLabor;
    vehicleCost += dVehicle;
    equipmentCost += dEquip;
    materialCostDiaries += dMaterial;
    directForOverhead += dLabor + dVehicle + dEquip + dMaterial;

    if (d.approvalStatus === "approved") hasApproved = true;
  }

  // Inventory material cost (read-only: catalog_items.cost_price)
  let materialCostInventory = 0;
  let hasMissingCostPrice = false;
  for (const c of consumptions) {
    if (c.costPrice != null && c.costPrice > 0) {
      materialCostInventory += c.quantity * c.costPrice;
    } else if (c.costPrice == null) {
      hasMissingCostPrice = true;
    }
  }

  const materialCost = materialCostDiaries + materialCostInventory;
  const overheadCost = diaries.length > 0
    ? directForOverhead * (rates.overheadPercentage / 100) + rates.fixedDailyOverhead * diaries.length
    : 0;
  const totalCost = laborCost + vehicleCost + equipmentCost + materialCost + overheadCost;
  const grossProfit = revenue - totalCost;
  const grossMarginPercent = revenue > 0
    ? (grossProfit / revenue) * 100
    : (totalCost > 0 ? -100 : 0);

  // Missing data tags
  const missingData: MissingDataTag[] = [];
  if (revenue === 0)                                             missingData.push("no_revenue");
  if (diaries.length === 0)                                      missingData.push("no_linked_diaries");
  else if (totalWorkers === 0)                                   missingData.push("no_crew_data");
  if (hasMissingCostPrice)                                       missingData.push("missing_cost_price");
  if (consumptions.length > 0 && materialCost === 0)             missingData.push("no_material_cost");
  if (diaries.length > 0 && !someHaveVehicle)                   missingData.push("no_vehicle_data");
  if (diaries.length > 0 && !hasApproved)                       missingData.push("no_approved_diary");

  // Confidence level
  let confidenceLevel: ConfidenceLevel;
  if (revenue === 0) {
    confidenceLevel = "missing_data";
  } else if (missingData.length === 0) {
    confidenceLevel = "high";
  } else if (missingData.includes("no_crew_data") || missingData.length >= 3) {
    confidenceLevel = "low";
  } else {
    confidenceLevel = "medium";
  }

  return {
    orderId: order.id,
    orderNumber: order.orderNumber,
    customerName: order.customer,
    revenue,
    laborCost,
    materialCost,
    vehicleCost,
    equipmentCost,
    subcontractorCost: 0,
    otherCost: 0,
    overheadCost,
    totalCost,
    grossProfit,
    grossMarginPercent,
    confidenceLevel,
    missingData,
    sourceData: {
      diaryIds: diaries.map(d => d.id),
      consumptionIds: consumptions.map(c => c.id),
      orderId: order.id,
    },
  };
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getServiceSupabase } from "@/lib/supabase/server";
import { verifyMasterAuth } from "@/lib/agents/scan-utils";

export async function GET(req: NextRequest) {
  const db = getServiceSupabase();
  const token = (req.headers.get("authorization") ?? "").replace(/^Bearer\s+/i, "");
  const userId = await verifyMasterAuth(db, token);
  if (!userId) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { data, error } = await db
    .from("profitability_snapshots")
    .select(`
      id, order_id, revenue, labor_cost, material_cost, vehicle_cost,
      equipment_cost, overhead_cost, total_cost, gross_profit,
      gross_margin_percent, confidence_level, missing_data, updated_at,
      work_orders!inner(order_number, customer, status, accounting_status)
    `)
    .is("work_diary_id", null)
    .order("gross_margin_percent", { ascending: true })
    .limit(300);

  if (error) return NextResponse.json({ error: error.message }, { status: 500 });
  return NextResponse.json(data ?? []);
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getServiceSupabase } from "@/lib/supabase/server";
import { verifyMasterAuth } from "@/lib/agents/scan-utils";
import {
  calculateOrderProfitability,
  type OrderInput,
  type OrderDiaryInput,
  type OrderConsumptionInput,
} from "@/lib/profitability";
import { DEFAULT_COST_RATES } from "@/types/costRates";
import type { CostRates } from "@/types/costRates";

export async function POST(req: NextRequest) {
  const db = getServiceSupabase();
  const token = (req.headers.get("authorization") ?? "").replace(/^Bearer\s+/i, "");
  const userId = await verifyMasterAuth(db, token);
  if (!userId) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    // Load cost rates
    const { data: ratesRow } = await db
      .from("cost_rates")
      .select("data")
      .eq("id", 1)
      .maybeSingle() as { data: { data: Record<string, unknown> } | null };
    const rates: CostRates = { ...DEFAULT_COST_RATES, ...((ratesRow?.data ?? {}) as Partial<CostRates>) };

    // Load orders in scope: completed status OR in accounting pipeline
    const { data: rawOrders, error: ordersErr } = await db
      .from("work_orders")
      .select("id,order_number,customer,status,accounting_status,billed_amount")
      .or("status.eq.completed,accounting_status.in.(verified,approved,invoiced,paid)")
      .neq("status", "cancelled")
      .order("created_at", { ascending: false })
      .limit(500);
    if (ordersErr) throw new Error(ordersErr.message);
    const orders = rawOrders ?? [];
    if (orders.length === 0) {
      return NextResponse.json({ generated: 0, updated: 0, missingData: 0, errors: [] });
    }

    const orderIds = orders.map(o => o.id as string);

    // Load all linked diaries in one query
    const { data: rawDiaries, error: diaryErr } = await db
      .from("work_diaries")
      .select("id,order_id,approval_status,data")
      .in("order_id", orderIds)
      .eq("status", "submitted");
    if (diaryErr) throw new Error(diaryErr.message);

    // Load consumptions (read-only)
    const { data: rawConsumptions, error: consErr } = await db
      .from("inventory_consumptions")
      .select("id,item_id,order_id,quantity,status")
      .in("order_id", orderIds)
      .in("status", ["consumed", "pending_review"]);
    if (consErr) throw new Error(consErr.message);

    // Load catalog items with cost_price for consumed items
    const consumedItemIds = [
      ...new Set((rawConsumptions ?? []).map(c => c.item_id as string)),
    ];
    const catalogCostMap = new Map<string, number | null>();
    if (consumedItemIds.length > 0) {
      const { data: catItems } = await db
        .from("catalog_items")
        .select("id,cost_price")
        .in("id", consumedItemIds);
      for (const ci of catItems ?? []) {
        catalogCostMap.set(
          ci.id as string,
          ci.cost_price != null ? Number(ci.cost_price) : null
        );
      }
    }

    // Load existing snapshots to avoid full re-insert
    const { data: existing } = await db
      .from("profitability_snapshots")
      .select("id,order_id")
      .in("order_id", orderIds)
      .is("work_diary_id", null);
    const existingMap = new Map<string, string>(); // order_id → snapshot id
    for (const s of existing ?? []) {
      existingMap.set(s.order_id as string, s.id as string);
    }

    // Build typed input arrays (shared across all orders)
    const diaryInputs: OrderDiaryInput[] = (rawDiaries ?? []).map(d => {
      const data = (d.data ?? {}) as Record<string, unknown>;
      return {
        id: d.id as string,
        orderId: d.order_id as string | null,
        billedAmount:      typeof data.billedAmount === "number" ? data.billedAmount : null,
        materialCost:      typeof data.materialCost === "number" ? data.materialCost : null,
        equipmentCost:     typeof data.equipmentCost === "number" ? data.equipmentCost : null,
        vehicleCostOverride: typeof data.vehicleCostOverride === "number" ? data.vehicleCostOverride : null,
        vehicleNumber:     (data.vehicleNumber as string) || null,
        crewLeaderName:    (data.crewLeaderName as string) || null,
        crewMembers:       Array.isArray(data.crewMembers) ? (data.crewMembers as string[]) : [],
        approvalStatus:    d.approval_status as string | null,
        isBillable:        typeof data.isBillable === "boolean" ? data.isBillable : null,
      };
    });

    const consumptionInputs: OrderConsumptionInput[] = (rawConsumptions ?? []).map(c => ({
      id:        c.id as string,
      orderId:   c.order_id as string,
      quantity:  Number(c.quantity),
      costPrice: catalogCostMap.get(c.item_id as string) ?? null,
    }));

    // Generate snapshot for each order
    let generated = 0;
    let updated = 0;
    let missingDataCount = 0;
    const errors: string[] = [];

    for (const order of orders) {
      try {
        const orderInput: OrderInput = {
          id:           order.id as string,
          orderNumber:  order.order_number as string,
          customer:     order.customer as string,
          billedAmount: order.billed_amount != null ? Number(order.billed_amount) : null,
        };

        const result = calculateOrderProfitability(orderInput, diaryInputs, rates, consumptionInputs);
        if (result.confidenceLevel === "missing_data") missingDataCount++;

        const snapshotRow = {
          order_id:            result.orderId,
          work_diary_id:       null,
          customer_id:         null,
          revenue:             result.revenue,
          labor_cost:          result.laborCost,
          material_cost:       result.materialCost,
          vehicle_cost:        result.vehicleCost,
          equipment_cost:      result.equipmentCost,
          subcontractor_cost:  result.subcontractorCost,
          other_cost:          result.otherCost,
```

Add the following component before the `ProfitabilityPage` export function:

```typescript
// ── CFO Snapshot row shape (from /api/profitability/snapshots) ─────────────

interface SnapshotRow {
  order_id: string;
  revenue: number;
  labor_cost: number;
  material_cost: number;
  vehicle_cost: number;
  overhead_cost: number;
  total_cost: number;
  gross_profit: number;
  gross_margin_percent: number;
  confidence_level: ConfidenceLevel;
  missing_data: MissingDataTag[];
  updated_at: string;
  work_orders: {
    order_number: string;
    customer: string;
    status: string;
    accounting_status: string;
  };
}

// ── CFO Lite tab ──────────────────────────────────────────────────────────────

function CfoTab() {
  const [snapshots, setSnapshots] = useState<SnapshotRow[]>([]);
  const [loading, setLoading] = useState(true);
  const [generating, setGenerating] = useState(false);
  const [filterConf, setFilterConf] = useState<ConfidenceLevel | "all">("all");
  const [lastGenerated, setLastGenerated] = useState<string | null>(null);
  const [genResult, setGenResult] = useState<{ generated: number; updated: number; missingData: number } | null>(null);
  const [error, setError] = useState<string | null>(null);

  const loadSnapshots = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const supabase = getSupabase();
      const { data: session } = await supabase.auth.getSession();
      const token = session?.session?.access_token;
      if (!token) { setError("נדרש אימות"); setLoading(false); return; }

      const res = await fetch("/api/profitability/snapshots", {
        headers: { Authorization: `Bearer ${token}` },
      });
      if (!res.ok) throw new Error(await res.text());
      const data = await res.json() as SnapshotRow[];
      setSnapshots(data);
      if (data.length > 0) setLastGenerated(data[0].updated_at);
    } catch (err) {
      setError(err instanceof Error ? err.message : "שגיאה בטעינה");
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { void loadSnapshots(); }, [loadSnapshots]);

  async function handleGenerate() {
    setGenerating(true);
    setGenResult(null);
    setError(null);
    try {
      const supabase = getSupabase();
      const { data: session } = await supabase.auth.getSession();
      const token = session?.session?.access_token;
      if (!token) { setError("נדרש אימות"); return; }

      const res = await fetch("/api/profitability/snapshots/generate", {
        method: "POST",
        headers: { Authorization: `Bearer ${token}` },
      });
      if (!res.ok) throw new Error(await res.text());
      const result = await res.json() as { generated: number; updated: number; missingData: number; errors: string[] };
      setGenResult({ generated: result.generated, updated: result.updated, missingData: result.missingData });
      await loadSnapshots();
    } catch (err) {
      setError(err instanceof Error ? err.message : "שגיאה ביצירת סנאפשוטים");
    } finally {
      setGenerating(false);
    }
  }

  const filtered = filterConf === "all"
    ? snapshots
    : snapshots.filter(s => s.confidence_level === filterConf);

  const totalRev    = snapshots.reduce((s, r) => s + r.revenue, 0);
  const totalProfit = snapshots.reduce((s, r) => s + r.gross_profit, 0);
  const lossCount   = snapshots.filter(s => s.gross_profit < 0 && s.revenue > 0).length;
  const noDataCount = snapshots.filter(s => s.confidence_level === "missing_data").length;
  const highCount   = snapshots.filter(s => s.confidence_level === "high").length;

  const CONF_LEVELS: ConfidenceLevel[] = ["high", "medium", "low", "missing_data"];

  return (
    <div className="space-y-4">
      {/* Header bar */}
      <div className="bg-white rounded-xl border border-gray-200 shadow-sm px-5 py-4 flex items-center justify-between flex-wrap gap-3">
        <div>
          <div className="text-sm font-bold text-gray-800">רווחיות עבודה — CFO ליי</div>
          <div className="text-xs text-gray-400">
            נתונים אנליטיים בלבד · אינם משפיעים על חיובים או חשבוניות
          </div>
        </div>
        <div className="flex items-center gap-3 flex-wrap">
          {lastGenerated && (
            <span className="text-[11px] text-gray-400">
              עודכן: {new Date(lastGenerated).toLocaleDateString("he-IL", { day: "2-digit", month: "2-digit", year: "2-digit", hour: "2-digit", minute: "2-digit" })}
            </span>
          )}
          {genResult && (
            <span className="text-[11px] text-green-600 font-medium">
              ✓ {genResult.generated} חדשים · {genResult.updated} עודכנו
            </span>
          )}
          <button
            type="button"
            onClick={() => void handleGenerate()}
            disabled={generating || loading}
            className="px-3 py-1.5 rounded-lg bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300 text-white text-xs font-bold transition-colors flex items-center gap-1.5"
          >
            {generating ? (
              <span className="inline-block w-3 h-3 border-2 border-white border-t-transparent rounded-full animate-spin" />
            ) : null}
            חשב מחדש
          </button>
        </div>
      </div>

      {error && (
        <div className="bg-red-50 border border-red-200 rounded-xl px-4 py-3 text-sm text-red-700">{error}</div>
      )}

      {/* KPI strip */}
      {snapshots.length > 0 && (
        <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
          <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
            <div className="h-1 bg-blue-500" />
            <div className="px-4 py-3">
              <div className="text-2xl font-black text-gray-900">{snapshots.length}</div>
              <div className="text-xs text-gray-500 mt-0.5">הזמנות</div>
            </div>
          </div>
          <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
            <div className="h-1 bg-green-500" />
            <div className="px-4 py-3">
              <div className="text-lg font-black text-gray-900" dir="ltr">
                {totalRev > 0 ? `₪${Math.round(totalRev).toLocaleString("he-IL")}` : "—"}
              </div>
              <div className="text-xs text-gray-500 mt-0.5">הכנסה כוללת</div>
            </div>
          </div>
          <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
            <div className={`h-1 ${totalProfit >= 0 ? "bg-emerald-500" : "bg-red-500"}`} />
            <div className="px-4 py-3">
              <div className={`text-lg font-black ${totalProfit >= 0 ? "text-emerald-700" : "text-red-700"}`} dir="ltr">
                {totalRev > 0 ? `${totalProfit >= 0 ? "+" : ""}₪${Math.round(Math.abs(totalProfit)).toLocaleString("he-IL")}` : "—"}
              </div>
              <div className="text-xs text-gray-500 mt-0.5">רווח גולמי</div>
            </div>
          </div>
          <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
            <div className="h-1 bg-red-500" />
            <div className="px-4 py-3">
              <div className="text-2xl font-black text-red-700">{lossCount}</div>
              <div className="text-xs text-gray-500 mt-0.5">הזמנות הפסדיות</div>
            </div>
          </div>
          <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
            <div className="h-1 bg-gray-300" />
            <div className="px-4 py-3">
              <div className="text-2xl font-black text-gray-500">{noDataCount}</div>
              <div className="text-xs text-gray-500 mt-0.5">חסרי נתונים</div>
            </div>
          </div>
        </div>
      )}

      {/* Confidence filter */}
      {snapshots.length > 0 && (
        <div className="bg-white rounded-xl border border-gray-200 shadow-sm px-4 py-3 flex items-center gap-2 flex-wrap">
          <span className="text-xs text-gray-500 font-medium">רמת ודאות:</span>
          <button type="button" onClick={() => setFilterConf("all")}
            className={`px-2.5 py-1 rounded-full text-xs font-medium transition-colors ${filterConf === "all" ? "bg-gray-800 text-white" : "bg-gray-100 text-gray-600 hover:bg-gray-200"}`}>
            הכל ({snapshots.length})
          </button>
          {CONF_LEVELS.map(cl => {
            const cnt = snapshots.filter(s => s.confidence_level === cl).length;
            if (cnt === 0) return null;
            return (
              <button key={cl} type="button" onClick={() => setFilterConf(cl)}
                className={`inline-flex items-center gap-1 px-2.5 py-1 rounded-full text-xs font-medium transition-colors ${filterConf === cl ? CONFIDENCE_COLORS[cl] + " ring-1 ring-current" : "bg-gray-100 text-gray-600 hover:bg-gray-200"}`}>
                {CONFIDENCE_LABELS[cl]} ({cnt})
              </button>
            );
          })}
        </div>
      )}

      {/* Snapshot table */}
      <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
        <div className="px-4 py-3 border-b border-gray-100 flex items-center justify-between">
          <span className="text-sm font-bold text-gray-700">פירוט רווחיות לפי הזמנה</span>
          <span className="text-xs text-gray-400">{filtered.length} הזמנות</span>
        </div>

        {/* Table header */}
        <div className="flex items-center gap-2 px-4 py-2 bg-gray-50 border-b border-gray-100 text-[10px] font-bold text-gray-400 uppercase tracking-wide">
          <div className="flex-1 min-w-0">הזמנה / לקוח</div>
          <div className="w-20 text-left shrink-0">הכנסה</div>
          <div className="w-18 text-left shrink-0">עלות עובדים</div>
          <div className="w-18 text-left shrink-0">עלות חומרים</div>
          <div className="w-18 text-left shrink-0">סה״כ עלות</div>
          <div className="w-20 text-left shrink-0">רווח גולמי</div>
          <div className="w-16 text-center shrink-0">%</div>
          <div className="w-20 shrink-0">ודאות</div>
          <div className="w-24 shrink-0">חסרים</div>
        </div>

        {loading ? (
          <div className="px-4 py-10 text-center text-sm text-gray-400">טוען נתונים...</div>
        ) : filtered.length === 0 ? (
          <div className="px-4 py-10 text-center">
            <p className="text-sm text-gray-400">
              {snapshots.length === 0
                ? "אין נתוני רווחיות. לחץ 'חשב מחדש' כדי לצור סנאפשוטים."
                : "אין הזמנות המתאימות לסינון."}
            </p>
          </div>
        ) : (
          filtered.map(s => {
            const isLoss = s.gross_profit < 0 && s.revenue > 0;
            const marginColor = isLoss
              ? "text-red-700"
              : s.gross_margin_percent < 12
              ? "text-amber-700"
              : "text-green-700";
            const o = s.work_orders;

            return (
              <div key={s.order_id} className={`flex items-center gap-2 px-4 py-2.5 border-b border-gray-50 hover:bg-gray-50 transition-colors text-xs ${isLoss ? "bg-red-50/40" : ""}`}>
                <div className="flex-1 min-w-0">
                  <div className="font-semibold text-gray-900 truncate">{o.order_number}</div>
                  <div className="text-[10px] text-gray-400 truncate">{o.customer}</div>
                </div>
                <div className="w-20 text-left text-gray-700 shrink-0" dir="ltr">
                  {s.revenue > 0 ? `₪${Math.round(s.revenue).toLocaleString("he-IL")}` : "—"}
                </div>
                <div className="w-18 text-left text-gray-600 shrink-0" dir="ltr">
                  {s.labor_cost > 0 ? `₪${Math.round(s.labor_cost).toLocaleString("he-IL")}` : "—"}
                </div>
                <div className="w-18 text-left text-gray-600 shrink-0" dir="ltr">
                  {s.material_cost > 0 ? `₪${Math.round(s.material_cost).toLocaleString("he-IL")}` : "—"}
                </div>
                <div className="w-18 text-left text-gray-600 shrink-0" dir="ltr">
                  {s.total_cost > 0 ? `₪${Math.round(s.total_cost).toLocaleString("he-IL")}` : "—"}
                </div>
                <div className={`w-20 text-left font-bold shrink-0 ${marginColor}`} dir="ltr">
                  {s.revenue > 0
                    ? `${isLoss ? "-" : "+"}₪${Math.round(Math.abs(s.gross_profit)).toLocaleString("he-IL")}`
                    : "—"}
                </div>
                <div className={`w-16 text-center font-bold shrink-0 text-xs ${marginColor}`}>
                  {s.revenue > 0 ? `${s.gross_margin_percent.toFixed(1)}%` : "—"}
                </div>
                <div className="w-20 shrink-0">
                  <span className={`inline-flex items-center text-[9px] font-bold px-1.5 py-0.5 rounded-full ${CONFIDENCE_COLORS[s.confidence_level]}`}>
                    {CONFIDENCE_LABELS[s.confidence_level]}
                  </span>
                </div>
                <div className="w-24 flex flex-wrap gap-0.5 shrink-0">
                  {(s.missing_data ?? []).slice(0, 2).map(tag => (
                    <span key={tag} className="text-[8px] bg-gray-100 text-gray-500 px-1 py-0.5 rounded">
                      {MISSING_DATA_LABELS[tag] ?? tag}
                    </span>
                  ))}
                  {(s.missing_data ?? []).length > 2 && (
                    <span className="text-[8px] text-gray-400">+{s.missing_data.length - 2}</span>
                  )}
                </div>
              </div>
            );
          })
        )}
      </div>

      <div className="text-[11px] text-gray-400 text-center">
        נתונים אנליטיים בלבד — אינם משפיעים על חיובים, חשבוניות, או רשומות כספיות
      </div>
    </div>
  );
}
```
